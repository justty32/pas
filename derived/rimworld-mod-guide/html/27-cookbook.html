使用前提 多數片段假設已 using Verse;、using RimWorld;、using Verse.AI;、using UnityEngine;。靜態入口幾乎都掛在 Find（執行期狀態）與 Current（遊戲/世界容器）之下。Find.CurrentMap 在世界地圖檢視時可能為 null，存取前請判空。
A. 取得常用物件
當前地圖（玩家正在看的那張，世界檢視時為 null）
Map map = Find.CurrentMap;
// 所有已載入的地圖
List<Map> maps = Find.Maps;
所有自由殖民者（含當前地圖 / 全派系）
// 當前地圖的自由殖民者
var colonists = map.mapPawns.FreeColonists;
// 跨地圖：所有玩家陣營的殖民者
var all = PawnsFinder.AllMaps_FreeColonists;
目前選取的物件（單選 / 多選）
Thing sel = Find.Selector.SingleSelectedThing;
var objs = Find.Selector.SelectedObjects;   // List<object>
var things = Find.Selector.SelectedObjectsListForReading;
滑鼠所在格（地圖座標 IntVec3）
IntVec3 cell = UI.MouseCell();
// 是否在地圖範圍內
bool valid = cell.InBounds(Find.CurrentMap);
當前派系 / 遊戲容器
Faction player = Faction.OfPlayer;
Game game = Current.Game;
World world = Find.World;
由 defName 取 Def（找不到回 null / 報錯）
var def = DefDatabase<ThingDef>.GetNamed("Steel");
var safe = DefDatabase<ThingDef>
    .GetNamedSilentFail("Steel");  // 找不到回 null
偏好 DefOf 而非字串查找 頻繁用到的 Def，宣告一個 [DefOf] public static class MyDefOf 讓框架在啟動時注入靜態欄位，比每次 GetNamed 快且型別安全。
B. 生成與移除
生成物品到地圖某格（含堆疊數量）
Thing steel = ThingMaker.MakeThing(ThingDefOf.Steel);
steel.stackCount = 75;
GenSpawn.Spawn(steel, cell, map);
生成帶材質的物品（如鋼製短刀）
Thing knife = ThingMaker.MakeThing(
    knifeDef, ThingDefOf.Steel);  // 第二參數 = stuff
GenSpawn.Spawn(knife, cell, map);
生成一名 Pawn（指定種族 + 派系）
var req = new PawnGenerationRequest(
    PawnKindDefOf.Colonist, Faction.OfPlayer);
Pawn pawn = PawnGenerator.GeneratePawn(req);
GenSpawn.Spawn(pawn, cell, map);
刪除 / 摧毀一個 Thing
thing.Destroy(DestroyMode.Vanish);  // 無殘骸
thing.Destroy(DestroyMode.KillFinalize);  // 走死亡流程
// 僅從地圖移除但保留物件
if (thing.Spawned) thing.DeSpawn();
掉落物品到附近（找鄰近可放格）
GenPlace.TryPlaceThing(
    thing, cell, map, ThingPlaceMode.Near);
讓 Pawn 從身上掉落物品（含轉成 unforbidden）
pawn.inventory.innerContainer.TryDrop(
    item, pawn.Position, map,
    ThingPlaceMode.Near, out Thing dropped);
生成後別忘記 Spawned 狀態 ThingMaker.MakeThing 只是建立物件、尚未在地圖上。沒呼叫 GenSpawn.Spawn 前 thing.Spawned 為 false，存取 thing.Map / thing.Position 會出錯。

D. 尋找與查詢
最近的某類物件（GenClosest 萬用查找）
Thing nearest = GenClosest.ClosestThingReachable(
    pawn.Position, map,
    ThingRequest.ForDef(ThingDefOf.Steel),
    PathEndMode.ClosestTouch,
    TraverseParms.For(pawn));
半徑內所有 Pawn（GenRadial 掃格）
var hits = GenRadial.RadialDistinctThingsAround(
    center, map, radius: 5f, useCenter: true)
    .OfType<Pawn>();
某格上的所有東西 / 第一個建築
var things = cell.GetThingList(map);   // List<Thing>
Building edifice = cell.GetEdifice(map);
Plant plant = cell.GetPlant(map);
可行走 / 可站立判斷
bool walkable = cell.Walkable(map);
bool standable = cell.Standable(map);  // 無阻擋物
bool reachable = pawn.CanReach(
    cell, PathEndMode.OnCell, Danger.Deadly);
全地圖某 Def 的物件清單（最快路徑）
var list = map.listerThings
    .ThingsOfDef(ThingDefOf.Steel);
// 依群組（如所有建築）
var blds = map.listerBuildings.allBuildingsColonist;
找一個落腳/可放置的格
CellFinder.TryFindRandomCellNear(
    center, map, radius: 4,
    c => c.Standable(map),
    out IntVec3 result);
效能：能用 lister 就別自己掃整張地圖 map.listerThings / map.listerBuildings 維護現成索引；對全圖 foreach 每格是 O(area)，在大地圖或 tick 迴圈裡會卡。需要可達性過濾時優先用 GenClosest（內建 BFS 早停）。

F. 時間與排程
當前 tick（遊戲時鐘，每秒約 60 tick）
int now = Find.TickManager.TicksGame;
// 絕對 tick（無視暫停，自開檔起算）
int abs = Find.TickManager.TicksAbs;
常用時間常數（GenDate）
int day  = GenDate.TicksPerDay;     // 60000
int hour = GenDate.TicksPerHour;    // 2500
int qtr  = GenDate.TicksPerQuadrum; // 一季
延遲執行（N tick 後跑一段程式）
long fireAt = Find.TickManager.TicksGame + 300;
Find.TickManager.later.ScheduleCallback(
    () => Log.Message("5 秒後執行"), 300);
tick 迴圈內的週期判斷（每 N tick 一次）
// 放在 CompTick / TickRare 裡
if (now.IsHashIntervalTick(250)) {
    // 每 250 tick 觸發一次（已自動錯開負載）
}
每日 / 每季判斷
if (now % GenDate.TicksPerDay == 0) { /* 每日 0 點 */ }
int dayOfQuadrum = GenLocalDate.DayOfQuadrum(map);
Season season = GenLocalDate.Season(map);
把 tick 數轉成可讀字串
string s = 90000.ToStringTicksToPeriod();  // "1.5 天"
string ago = (now - past).ToStringTicksToDays();
用 IsHashIntervalTick 自動分散負載 thing.IsHashIntervalTick(N)（Thing 的擴充）會依物件 hash 錯開觸發幀，避免所有物件在同一 tick 一起跑造成卡頓。比手寫 tick % N == 0 更好。
G. 存檔 (Scribe 速查)

所有 ExposeData() / PostExposeData() 內都用 Scribe_*。同一段程式碼同時負責「存」與「讀」，框架依 Scribe.mode 決定方向。

資料型別呼叫
基本值 (int/float/bool/string/enum)Scribe_Values.Look(ref x, "x", defaultVal);
指向已存在的 Thing/Pawn（存 ID）Scribe_References.Look(ref pawn, "pawn");
擁有的子物件（深度序列化）Scribe_Deep.Look(ref comp, "comp");
Def 參照Scribe_Defs.Look(ref myDef, "myDef");
IntVec3 / 自訂 structScribe_Values.Look(ref pos, "pos");
List<基本值>Scribe_Collections.Look(ref list, "list", LookMode.Value);
List<Thing 參照>Scribe_Collections.Look(ref things, "things", LookMode.Reference);
List<深度物件>Scribe_Collections.Look(ref deeps, "deeps", LookMode.Deep);
Dictionary<K,V>Scribe_Collections.Look(ref dict, "dict", LookMode.Value, LookMode.Value);
標準 ExposeData 樣板
public override void ExposeData() {
    base.ExposeData();
    Scribe_Values.Look(ref count, "count", 0);
    Scribe_References.Look(ref owner, "owner");
}
讀檔後修補（防 null、重建快取）
if (Scribe.mode == LoadSaveMode.PostLoadInit) {
    list ??= new List<Thing>();
    RebuildCache();
}
References vs Deep 不能搞錯 Scribe_References 用於「地圖/存檔已有的 Thing」（只存 loadID，讀檔後重連）；Scribe_Deep 用於「完全由你擁有的子物件」（連同欄位整包存）。把已 spawn 的 Thing 用 Deep 存會造成存檔損壞或物件重複。
H. 反射捷徑（Harmony AccessTools / Traverse）

需要存取私有成員時用 Harmony 的工具，比手刻 BindingFlags 簡潔。需 using HarmonyLib;。

快取一個私有欄位的 FieldInfo（高效，建議靜態快取）
static readonly FieldInfo f_jobs =
    AccessTools.Field(typeof(Pawn), "jobs");
// 讀值
var jobs = (Pawn_JobTracker)f_jobs.GetValue(pawn);
取私有方法並呼叫
var m = AccessTools.Method(
    typeof(SomeClass), "DoPrivate",
    new[] { typeof(int) });
m.Invoke(instance, new object[] { 42 });
Traverse 鏈式存取（一次性、可讀性高）
int val = Traverse.Create(obj)
    .Field("_inner")
    .Property("Count")
    .GetValue<int>();
Traverse 寫私有欄位 / 呼叫私有方法
Traverse.Create(pawn)
    .Field("mindState")
    .SetValue(newMindState);
Traverse.Create(obj)
    .Method("Recalc", args).GetValue();
Traverse 慢、AccessTools 快 Traverse 每次呼叫都做反射查找，適合冷路徑或一次性操作。tick 迴圈等熱路徑請改用快取的 AccessTools.Field/Method，或用 AccessTools.FieldRefAccess<T,F> 取得近乎原生速度的 getter/setter。
I. 數學與隨機
隨機數值（Rand，受存檔 seed 控制可重現）
int i = Rand.Range(1, 10);     // [1,10) int
int ii = Rand.RangeInclusive(1, 6); // [1,6] 含上界
float f = Rand.Range(0f, 1f);   // [0,1] float
機率判定與隨機挑選
if (Rand.Chance(0.3f)) { /* 30% 機率 */ }
var pick = list.RandomElement();
var w = list.RandomElementByWeight(x => x.weight);
IntVec3 四 / 八鄰格
foreach (IntVec3 adj in GenAdj.CardinalDirections)
    { var c = cell + adj; }   // 上下左右
// 含對角的 8 格
foreach (IntVec3 adj in GenAdj.AdjacentCells)
    { var c = cell + adj; }
半徑內所有格（GenRadial）
foreach (IntVec3 c in
    GenRadial.RadialCellsAround(center, 3.5f, true))
{
    if (c.InBounds(map)) { /* ... */ }
}
距離計算
float d  = a.DistanceTo(b);          // 歐氏距離
int d2  = a.DistanceToSquared(b);     // 平方（比大小用，免開根）
bool near = a.InHorDistOf(b, 5f);   // 水平距離內
數值夾擠 / 插值（Unity Mathf + GenMath）
float c = Mathf.Clamp(v, 0f, 1f);
float l = Mathf.Lerp(a, b, t);
float r = GenMath.RoundTo(v, 0.5f);  // 取最近 0.5 倍數
需要可重現的隨機？ 多人/驗證時用 Rand.PushState(seed) … Rand.PopState() 包住一段，確保同 seed 產生同結果且不污染全域隨機流。
